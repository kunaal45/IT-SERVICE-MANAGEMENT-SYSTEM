5️⃣ FIX FRONTEND (NEXT PRIORITY)
═════════════════════════════════════════════════════════════════════════════

Remove hardcoded data from:
  - admin-dashboard.html
  - engineer-dashboard.html
  - faculty-dashboard.html

Replace with API calls.

Example admin-dashboard.html:

  <script>
    async function loadDashboard() {
      const token = localStorage.getItem('token');
      const response = await fetch('/api/dashboard/admin', {
        headers: { 'Authorization': 'Bearer ' + token }
      });
      const data = await response.json();
      
      // Replace hardcoded numbers with real data
      document.getElementById('totalTickets').innerText = data.data.totalTickets;
      document.getElementById('openTickets').innerText = data.data.openTickets;
      document.getElementById('assignedTickets').innerText = data.data.assignedTickets;
      document.getElementById('inProgressTickets').innerText = data.data.inProgressTickets;
      document.getElementById('resolvedTickets').innerText = data.data.resolvedTickets;
      document.getElementById('closedTickets').innerText = data.data.closedTickets;
    }
    
    document.addEventListener('DOMContentLoaded', loadDashboard);
  </script>
